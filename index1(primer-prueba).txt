// Importa la librería Express del paquete instalado en node_modules.
// Express es una herramienta para crear servidores web en Node.js.
import express from 'express';

// Importa el contenido del archivo bd.json como un valor JSON.
// El objeto `datos` contendrá la lista de personas definida en bd.json.
import datos from './bd.json' with { type: 'json' };

// Crea la aplicación Express.
// `app` es un objeto que nos permite definir rutas y escuchar solicitudes.
const app = express();

// Define una ruta HTTP GET en '/'.
// Esta función se ejecuta cada vez que un cliente pide la dirección raíz.
app.get('/', (req, res) => {
    // `req` representa la solicitud que hace el cliente.
    // `res` es el objeto que usamos para responder con algo al cliente.

    // Envía los datos importados de bd.json en formato JSON.
    // El cliente recibe un JSON con la lista de personas.
    res.json(datos);
});

// Inicia el servidor en el puerto 3000.
// El primer argumento es el número de puerto.
// El segundo argumento es una función que se ejecuta cuando el servidor ya está listo.
app.listen(3000, () => {
    // Muestra un mensaje en la consola para saber que el servidor arrancó correctamente.
    console.log('El servidor está corriendo en el puerto 3000');
});
